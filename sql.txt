CREATE TABLE empleados (
    id SERIAL PRIMARY KEY,
    id_biometrico VARCHAR(10) UNIQUE NOT NULL,
    nombre_completo VARCHAR(150) NOT NULL,
    tipo_documento VARCHAR(20) DEFAULT 'DNI', -- DNI o Carnet Extranjería
    numero_documento VARCHAR(20) UNIQUE NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabla para los 4 estados (Ingreso, Refri, Salida)
CREATE TABLE marcaciones (
    id SERIAL PRIMARY KEY,
    id_biometrico VARCHAR(10) REFERENCES empleados(id_biometrico),
    fecha_hora TIMESTAMP NOT NULL,
    estado_calculado VARCHAR(30) DEFAULT 'PENDIENTE', -- Aquí se guardará si es Ingreso, Refri, etc.
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Creamos un tipo personalizado para los estados
CREATE TYPE tipo_marcacion AS ENUM ('INGRESO', 'SALIDA_REFRIGERIO', 'INGRESO_REFRIGERIO', 'SALIDA', 'OTRO');

CREATE TABLE empleados (
    id SERIAL PRIMARY KEY,
    id_biometrico VARCHAR(10) UNIQUE NOT NULL,
    nombre_completo VARCHAR(150) NOT NULL,
    tipo_documento VARCHAR(20) CHECK (tipo_documento IN ('DNI', 'CARNET_EXTRANJERIA')) DEFAULT 'DNI',
    numero_documento VARCHAR(20) UNIQUE NOT NULL,
    fecha_creacion TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE marcaciones (
    id SERIAL PRIMARY KEY,
    id_biometrico VARCHAR(10) REFERENCES empleados(id_biometrico) ON DELETE CASCADE,
    fecha_hora TIMESTAMPTZ NOT NULL,
    estado_calculado tipo_marcacion DEFAULT 'INGRESO',
    fecha_registro TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- Índices para optimizar búsquedas de reportes
CREATE INDEX idx_marcaciones_biometrico ON marcaciones(id_biometrico);
CREATE INDEX idx_marcaciones_fecha ON marcaciones(fecha_hora);
